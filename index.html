動態樣式生成函數


// 根據卡牌屬性和稀有度動態生成 Tailwind Classes
function generateCardClasses(attribute, rarity) {
  const baseClasses = 'card-base rounded-2xl p-4 transition-all duration-300 hover:-translate-y-2';
  
  // 屬性色彩映射
  const attributeColors = {
    human: { base: 'human-base', dark: 'human-dark', text: 'human-text' },
    yin: { base: 'yin-base', dark: 'yin-dark', text: 'yin-text' },
    yang: { base: 'yang-base', dark: 'yang-dark', text: 'yang-text' }
  };
  
  // 稀有度質感映射
  const rarityStyles = {
    common: (colors) => `bg-${colors.base}`,
    rare: (colors) => `bg-gradient-to-br from-${colors.base} to-${colors.dark} shadow-md`,
    legendary: (colors) => `bg-gradient-to-br from-${colors.base} to-${colors.dark} shadow-lg ring-2 ring-legendary-ring`
  };
  
  const colors = attributeColors[attribute];
  const backgroundStyle = rarityStyles[rarity](colors);
  
  return `${baseClasses} ${backgroundStyle} text-${colors.text}`;
}

// 使用示例
const cardClass = generateCardClasses('human', 'legendary');
// 結果: "card-base rounded-2xl p-4 ... bg-gradient-to-br from-human-base to-human-dark shadow-lg ring-2 ring-legendary-ring text-human-text"

React 組件範例


function Card({ card }) {
  const cardClasses = generateCardClasses(card.attribute, card.rarity);
  
  return (
    <div className={cardClasses}>
      <div className="text-center mb-4">
        <div className="bg-black/20 text-xs px-2 py-1 rounded inline-block mb-2">
          {card.rarity}
        </div>
        <h4 className="text-lg font-bold">{card.name}</h4>
      </div>
      
      <div className="bg-black/20 h-32 rounded-lg mb-4 flex items-center justify-center">
        <img src={card.artwork} alt={card.name} className="max-w-full max-h-full object-cover rounded" />
      </div>
      
      <div className="text-center space-y-2">
        <div className="text-sm">{card.description}</div>
        <div className="flex justify-around">
          <span>⚔️ {card.attack}</span>
          <span>💎 {card.crit}</span>
        </div>
      </div>
    </div>
  );
}

📋 設計規範總結

✅ 設計優勢

  • • 文化內涵深度：和色系統賦予卡牌東方美學
  • • 視覺統一性：所有卡牌保持和諧的色彩基調
  • • 工程友好性：Tailwind CSS 原子類易於實現
  • • 可擴展性：新屬性只需添加對應色彩配置
  • • 性能優化：純 CSS 實現，無需額外資源

🚀 改進建議

  • • 動效增強：添加更豐富的 hover 和點擊動效
  • • 無障礙支持：考慮色盲用戶的區分度
  • • 響應式適配：不同螢幕尺寸的卡牌大小調整
  • • 主題切換：支持深色/淺色模式切換
  • • 個性化：允許玩家自定義卡背或邊框
